# Variables de Entorno - Ejemplo
# Copia este archivo a .env.local y completa con tus credenciales reales

# Supabase Configuration
# Obtén estas credenciales desde tu proyecto en https://supabase.com/dashboard
NEXT_PUBLIC_SUPABASE_URL=https://your-project.supabase.co
NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY=your-publishable-key-here

# Service Role Key (solo para operaciones del servidor, nunca exponer en el cliente)
# Usar solo en Server Actions o API Routes
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# Application URL
# En desarrollo: http://localhost:3000
# En producción: https://tu-dominio.vercel.app
NEXT_PUBLIC_APP_URL=http://localhost:3000
